% Phantom Codes — supplementary materials master file (JAMIA supplement).

% instead of sections/*.tex.
%
% Build:
%     cd paper && make phantom_codes_supp
%
% Output: paper/phantom_codes/build/supplementary.pdf (snapshotted to
%         paper/phantom_codes_supplementary.pdf via
%         `make snapshot-phantom_codes-supp`).
%
% Editing model:
%   - Prose edits happen in supp_sections/*.tex.
%   - This file only changes when you add/remove/reorder sections,
%     change document class, or restyle the title.

% Tables in this supplement are wide auditability tables (28 configs ×
% 4 modes × 6 buckets, full Wilson CIs, etc.). Under the default
% \doublespacing applied above, those tables break visually: long
% model identifiers like ``claude-haiku-4-5:zeroshot'' wrap across
% multiple lines and adjacent cells collide. Fix: every longtable
% body renders single-spaced and at \footnotesize; the \doublespacing
% setting is restored automatically when the longtable closes.
\AtBeginEnvironment{longtable}{\singlespacing\footnotesize}

\usepackage{amsmath}
\usepackage{amssymb}

% biblatex with biber backend — matches main.tex; same Vancouver style
